class Dog {
  #name;
  #size;
  constructor(name, size) {
    this.#name = name;
    if (size !== 'S' || size !== 'M' || size !== 'L') {
      throw new Error('올바른 사이즈를 입력하세요.');
    } else {
      this.#size = size;
    }
  }

  get name() {
    return this.#name;
  }

  get bathingFee() {
    if (this.#size === 'S') {
      return 40000;
    } else if (this.#size === 'M') {
      return 50000;
    } else {
      return 60000;
    }
  }
}

const smallSizeDog = new Dog('dog1', 'S');
const mediumSizeDog = new Dog('dog2', 'M');
const LargeSizeDog = new Dog('dog3', 'L');
