## 📌문제1
### 다음 코드의 동일한 기능을 하도록 리팩터링 하시오.
```js
class Book {
  #title;
  #writer;
  #price;
  #discount;
  #calculatedPrice;

  constructor(title, name, phoneNumber, price) {
    this.#title = title;
    this.#writer = new Writer(name, phoneNumber);
    this.#price = price;
    this.#discount = 0;
  }

  get title() {
    return this.#title;
  }

  get writerName() {
    return this.#writer.name;
  }

  set writerName(arg) {
    this.#writer.name = arg;
  }

  get writerPhoneNumber() {
    return this.#writer.phoneNumber;
  }

  set writerPhoneNumber(arg) {
    this.#writer.phoneNumber = arg;
  }

  set price(arg) {
    this.#price = arg;
    this.#calculatedPrice = this.#price - this.#discount;
  }

  set discount(arg) {
    this.#discount = arg;
    this.#calculatedPrice = this.#price - this.#discount;
  }

  get calculatedPrice() {
    return this.#calculatedPrice;
  }
}

class Writer {
  #name;
  #phoneNumber;

  constructor(name, phoneNumber) {
    this.#name = name;
    this.#phoneNumber = phoneNumber;
  }

  get name() {
    return this.#name;
  }

  set name(arg) {
    this.#name = arg;
  }

  get phoneNumber() {
    return this.#phoneNumber;
  }

  set phoneNumber(arg) {
    this.#phoneNumber = arg;
  }
}
```

## 📌문제2
### 다음 코드의 출력 결과가 동일하도록 리팩터링 하시오.
```js
class App {
  name;
  type;
  downloads;
  platform;
  constructor(name, type, downloads, platform) {
    this.name = name;
    this.type = type;
    this.downloads = downloads;
    this.platform = platform;
  }

  get name() {
    return this.name;
  }

  get type() {
    return this.type;
  }

  get downloads() {
    return this.downloads;
  }

  get platform() {
    return this.platform;
  }
}

const apps = [
  new App("테트리스", "Puzzle", 486, "Play-Store"),
  new App("배틀그라운드", "FPS", 9999999, "App-Store"),
  new App("던전앤파이터", "RPG", 9505, "Play-Store"),
  new App("디데이 계산기", "etc", 12, "App-Store"),
  new App(null, null, null, null),
];

apps.forEach((app) =>
  console.log([
    app.name || "미분류",
    getType(app),
    degreeOfPopular(app.downloads),
    app.platform || "미등록",
  ])
);

function getType(app) {
  if (app.platform === "Play-Store") {
    switch (app.type) {
      case "RPG":
      case "FPS":
      case "Puzzle":
        return app.type;
      default:
        return "etc";
    }
  } else if (app.platform === "App-Store") {
    switch (app.type) {
      case "RPG":
        return "롤플레잉";
      case "FPS":
        return "슈팅";
      case "Puzzle":
        return "전략";
      default:
        return "etc";
    }
  } else {
    return "미분류";
  }
}

function degreeOfPopular(downloads) {
  if (downloads === null) {
    return "측정 불가";
  }
  if (10000 < downloads) return "OMG";
  else if (1000 < downloads) return "Hot";
  else if (100 < downloads) return "Rising";
  else return "New";
}
```
