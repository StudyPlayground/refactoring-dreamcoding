    return this._count;
  }

  set count(arg) {

  }

  }

function recruitingState(count) {
  if (10 < count) {
    return "Full";
  } else if (5 < count) {
    return "Almost Full";
  } else {
    return "Hiring";
  }
}

  return study1.count - study2.count;
}
